# Customization

Tailwind CSS is designed to be highly customizable. You can extend or override the default configuration to match your project's design system perfectly. The main configuration file is `tailwind.config.js`.

---

## 1. Basic Configuration File

When you run `npx tailwindcss init`, it creates a basic configuration file:

```javascript
/** @type {import('tailwindcss').Config} */
module.exports = {
  content: [
    "./src/**/*.{html,js,jsx,ts,tsx}",
  ],
  theme: {
    extend: {},
  },
  plugins: [],
}
```

### Extended Configuration Example:
```javascript
/** @type {import('tailwindcss').Config} */
module.exports = {
  content: [
    "./src/**/*.{html,js,jsx,ts,tsx}",
    "./public/index.html",
  ],
  darkMode: 'class', // or 'media'
  theme: {
    extend: {
      colors: {
        brand: {
          primary: '#1E40AF',
          secondary: '#DC2626',
        }
      },
      fontFamily: {
        sans: ['Inter', 'system-ui', 'sans-serif'],
      },
    },
  },
  plugins: [
    require('@tailwindcss/forms'),
    require('@tailwindcss/typography'),
  ],
}
```

## 2. Color Customization

### Extending Colors (Recommended)
```javascript
module.exports = {
  theme: {
    extend: {
      colors: {
        // Add new colors
        brand: {
          50: '#f0f9ff',
          100: '#e0f2fe',
          200: '#bae6fd',
          300: '#7dd3fc',
          400: '#38bdf8',
          500: '#0ea5e9',
          600: '#0284c7',
          700: '#0369a1',
          800: '#075985',
          900: '#0c4a6e',
          950: '#082f49',
        },
        // Single custom color
        'custom-orange': '#FF6B35',
        // Using CSS variables
        'primary': 'var(--color-primary)',
        'secondary': 'var(--color-secondary)',
      },
    },
  },
}
```

### Overriding Colors (Replace Defaults)
```javascript
module.exports = {
  theme: {
    colors: {
      // Completely replace the default color palette
      transparent: 'transparent',
      current: 'currentColor',
      black: '#000',
      white: '#fff',
      gray: {
        50: '#f9fafb',
        100: '#f3f4f6',
        // ... custom gray shades
      },
      // Your custom colors
      primary: {
        500: '#3B82F6',
        600: '#2563EB',
      },
    },
  },
}
```

**Usage:**
```html
<div class="bg-brand-500 text-white p-4">Brand Color</div>
<button class="bg-custom-orange text-white px-4 py-2">Custom Orange</button>
```

## 3. Typography Customization

### Font Families
```javascript
module.exports = {
  theme: {
    extend: {
      fontFamily: {
        // Add new font families
        'display': ['"SF Pro Display"', 'system-ui', 'sans-serif'],
        'body': ['"Open Sans"', 'system-ui', 'sans-serif'],
        'mono': ['"JetBrains Mono"', 'monospace'],
        // Override default sans-serif
        'sans': ['Inter', 'system-ui', 'sans-serif'],
      },
    },
  },
}
```

**Usage:**
```html
<h1 class="font-display text-4xl">Display Font</h1>
<p class="font-body text-base">Body font for paragraphs</p>
<code class="font-mono text-sm">Code snippets</code>
```

### Font Sizes
```javascript
module.exports = {
  theme: {
    extend: {
      fontSize: {
        // Add custom sizes
        '2xs': ['0.625rem', { lineHeight: '0.75rem' }],
        '10xl': ['10rem', { lineHeight: '1' }],
        // Responsive font size example
        'fluid': 'clamp(1rem, 5vw, 3rem)',
      },
    },
  },
}
```

**Usage:**
```html
<p class="text-2xs">Extra extra small text</p>
<h1 class="text-10xl">Huge heading</h1>
<h2 class="text-fluid">Responsive heading</h2>
```

## 5. Breakpoints Customization

### Custom Breakpoints
```javascript
module.exports = {
  theme: {
    screens: {
      // Custom names
      'phone': '375px',
      'tablet': '768px',
      'laptop': '1024px',
      'desktop': '1440px',
      'tv': '1920px',
      
      // Or extend default
      'xs': '475px',
      ...defaultTheme.screens,
    },
  },
}
```

### Mobile-First Breakpoints
```javascript
module.exports = {
  theme: {
    screens: {
      'sm': '640px',
      'md': '768px',
      'lg': '1024px',
      'xl': '1280px',
      '2xl': '1536px',
      // Custom min-width breakpoint
      'portrait': { 'raw': '(orientation: portrait)' },
      // Max-width breakpoint
      'max-lg': { 'max': '1023px' },
      // Range
      'tablet-only': { 'min': '640px', 'max': '767px' },
    },
  },
}
```

**Usage:**
```html
<div class="phone:hidden tablet:block">Visible on tablet+</div>
<div class="portrait:flex landscape:hidden">Only in portrait</div>
<div class="max-lg:hidden">Hidden on small screens</div>
```

## 8. Creating Custom Utilities

### Using `@layer` Directives
Create a CSS file (e.g., `src/styles/tailwind.css`):

```css
@tailwind base;
@tailwind components;
@tailwind utilities;

@layer base {
  /* Custom base styles */
  h1 {
    @apply text-4xl font-bold;
  }
  
  /* CSS Custom Properties */
  :root {
    --color-primary: #3B82F6;
    --color-secondary: #10B981;
  }
}

@layer components {
  /* Custom component classes */
  .btn-primary {
    @apply px-4 py-2 bg-blue-500 text-white font-semibold rounded-lg 
           hover:bg-blue-600 focus:outline-none focus:ring-2 
           focus:ring-blue-300 transition-colors duration-200;
  }
  
  .card {
    @apply bg-white rounded-xl shadow-lg p-6;
  }
  
  .input-field {
    @apply w-full px-3 py-2 border border-gray-300 rounded-lg 
           focus:outline-none focus:ring-2 focus:ring-blue-500 
           focus:border-transparent transition-shadow;
  }
}

@layer utilities {
  /* Custom utility classes */
  .scrollbar-hide {
    -ms-overflow-style: none;
    scrollbar-width: none;
  }
  .scrollbar-hide::-webkit-scrollbar {
    display: none;
  }
  
  .text-shadow {
    text-shadow: 2px 2px 4px rgba(0,0,0,0.1);
  }
  
  .animate-float {
    animation: float 3s ease-in-out infinite;
  }
  
  @keyframes float {
    0%, 100% { transform: translateY(0px); }
    50% { transform: translateY(-10px); }
  }
  
  /* Responsive utilities */
  .text-balance {
    text-wrap: balance;
  }
}
```

### Component Extraction with `@apply`
```css
/* Extract common button patterns */
.btn {
  @apply inline-flex items-center justify-center font-medium 
         rounded-lg transition-colors focus:outline-none 
         focus:ring-2 focus:ring-offset-2;
}

.btn-sm {
  @apply px-3 py-1.5 text-sm;
}

.btn-md {
  @apply px-4 py-2;
}

.btn-lg {
  @apply px-6 py-3 text-lg;
}

.btn-primary {
  @apply bg-blue-600 text-white hover:bg-blue-700 focus:ring-blue-500;
}

.btn-secondary {
  @apply bg-gray-200 text-gray-800 hover:bg-gray-300 focus:ring-gray-400;
}
```

## 9. Dynamic Classes with JavaScript

### React Component with Dynamic Classes
```jsx
// Button.jsx
export default function Button({ 
  variant = 'primary', 
  size = 'md', 
  isLoading,
  children,
  ...props 
}) {
  const baseClasses = "font-medium rounded-lg transition-colors focus:outline-none focus:ring-2";
  
  const variants = {
    primary: "bg-blue-600 text-white hover:bg-blue-700 focus:ring-blue-500",
    secondary: "bg-gray-200 text-gray-800 hover:bg-gray-300 focus:ring-gray-400",
    danger: "bg-red-600 text-white hover:bg-red-700 focus:ring-red-500",
  };
  
  const sizes = {
    sm: "px-3 py-1.5 text-sm",
    md: "px-4 py-2",
    lg: "px-6 py-3 text-lg",
  };
  
  const loadingClasses = isLoading ? "opacity-70 cursor-not-allowed" : "";
  
  return (
    <button
      className={`${baseClasses} ${variants[variant]} ${sizes[size]} ${loadingClasses}`}
      disabled={isLoading}
      {...props}
    >
      {isLoading ? (
        <span className="flex items-center">
          <svg className="animate-spin -ml-1 mr-2 h-4 w-4" fill="none" viewBox="0 0 24 24">
            {/* Spinner SVG */}
          </svg>
          Loading...
        </span>
      ) : (
        children
      )}
    </button>
  );
}
```

### Vue Component with Dynamic Classes
```vue
<template>
  <button
    :class="[
      'font-medium rounded-lg transition-colors focus:outline-none focus:ring-2',
      variantClasses[variant],
      sizeClasses[size],
      isLoading ? 'opacity-70 cursor-not-allowed' : ''
    ]"
    :disabled="isLoading"
  >
    <slot />
  </button>
</template>

<script setup>
import { defineProps } from 'vue';

const props = defineProps({
  variant: {
    type: String,
    default: 'primary',
    validator: (value) => ['primary', 'secondary', 'danger'].includes(value)
  },
  size: {
    type: String,
    default: 'md',
    validator: (value) => ['sm', 'md', 'lg'].includes(value)
  },
  isLoading: Boolean
});

const variantClasses = {
  primary: 'bg-blue-600 text-white hover:bg-blue-700 focus:ring-blue-500',
  secondary: 'bg-gray-200 text-gray-800 hover:bg-gray-300 focus:ring-gray-400',
  danger: 'bg-red-600 text-white hover:bg-red-700 focus:ring-red-500'
};

const sizeClasses = {
  sm: 'px-3 py-1.5 text-sm',
  md: 'px-4 py-2',
  lg: 'px-6 py-3 text-lg'
};
</script>
```

## 10. Using CSS Variables

```javascript

  theme: {
    extend: {
      colors: {
        primary: 'var(--color-primary)',
```

## 11. Plugin Integration

### Official Plugins
```bash
npm install @tailwindcss/forms @tailwindcss/typography @tailwindcss/aspect-ratio
```

```javascript
// tailwind.config.js
module.exports = {
  plugins: [
    require('@tailwindcss/forms'),       // Better form styling
    require('@tailwindcss/typography'),  // Prose styles
    require('@tailwindcss/aspect-ratio'), // Aspect ratio utilities
    require('@tailwindcss/line-clamp'),  // Line clamping
  ],
}
```

### Custom Plugin Creation
```javascript
// tailwind.config.js
const plugin = require('tailwindcss/plugin');

module.exports = {
  plugins: [
    plugin(function({ addUtilities, addComponents, addBase, theme }) {
      // Add custom utilities
      addUtilities({
        '.scrollbar-hide': {
          /* Firefox */
          'scrollbar-width': 'none',
          /* IE and Edge */
          '-ms-overflow-style': 'none',
        },
        '.scrollbar-hide::-webkit-scrollbar': {
          display: 'none',
        },
        '.text-balance': {
          'text-wrap': 'balance',
        },
      });
      
      // Add custom components
      addComponents({
        '.btn': {
          padding: `${theme('spacing.2')} ${theme('spacing.4')}`,
          borderRadius: theme('borderRadius.md'),
          fontWeight: theme('fontWeight.medium'),
        },
        '.btn-primary': {
          backgroundColor: theme('colors.blue.600'),
          color: theme('colors.white'),
          '&:hover': {
            backgroundColor: theme('colors.blue.700'),
          },
        },
      });
      
      // Add base styles
      addBase({
        'h1': {
          fontSize: theme('fontSize.2xl'),
          fontWeight: theme('fontWeight.bold'),
        },
        'h2': {
          fontSize: theme('fontSize.xl'),
          fontWeight: theme('fontWeight.semibold'),
        },
      });
    }),
  ],
}
```

```javascript
module.exports = {
  content: [
    './src/**/*.{html,js,jsx,ts,tsx,vue}',
    './public/**/*.html',

  ],
  safelist: [

    'text-red-500',

    'text-(xs|sm|base|lg|xl)',
    // Dynamic classes

    'opacity-50',
    'opacity-100',
  ],
  // Or use blocklist to exclude specific classes
  blocklist: [
    'container',
    'mx-auto',
  ],
```

```javascript
// tailwind.preset.js
module.exports = {
  theme: {
    extend: {
      colors: {
        brand: {
          primary: '#1E40AF',

        }
      },

        sans: ['Inter', 'system-ui', 'sans-serif'],
      },

    require('@tailwindcss/forms'),
```

**Usage in main config:**
```javascript
// tailwind.config.js

  presets: [preset],

    },
  },
}
```

## 15. Best Practices for Customization

### Do:
1. **Use `extend`** for adding new values instead of overriding
2. **Create design tokens** for colors, spacing, etc.
3. **Use meaningful names** for custom values
4. **Keep custom CSS minimal** - prefer configuration over CSS
5. **Document custom utilities** and components
6. **Use presets** for multi-project consistency

### Don't:
1. **Don't override entire sections** unless necessary
2. **Avoid deep nesting** in configuration
3. **Don't create too many custom utilities** - use Tailwind's defaults when possible
4. **Avoid hard-coded values** in custom CSS when they could be in config

### Organization Example:
```javascript
// tailwind.config.js
const defaultTheme = require('tailwindcss/defaultTheme');

module.exports = {
  content: ['./src/**/*.{html,js,jsx,ts,tsx}'],
  
  theme: {
    // Extend rather than override
    extend: {
      // Colors
      colors: {
        brand: {
          light: '#f0f9ff',
          DEFAULT: '#0ea5e9',
          dark: '#0369a1',
        },
      },
      
      // Typography
      fontFamily: {
        sans: ['Inter var', ...defaultTheme.fontFamily.sans],
      },
      
      // Spacing
      spacing: {
        '18': '4.5rem',
        '128': '32rem',
      },
      
      // Animations
      animation: {
        'bounce-slow': 'bounce 2s infinite',
        'spin-slow': 'spin 3s linear infinite',
      },
    },
  },
  
  // Variants
  variants: {
    extend: {
      opacity: ['disabled'],
      scale: ['active'],
    },
  },
  
  // Plugins
  plugins: [
    require('@tailwindcss/forms'),
    // Custom plugin
    function({ addUtilities }) {
      addUtilities({
        '.text-shadow': {
          'text-shadow': '2px 2px 4px rgba(0,0,0,0.1)',
        },
      });
    },
  ],
};
```

## Summary

Tailwind CSS customization allows you to:

1. **Extend the default theme** with your design tokens
2. **Create custom utilities** and components
3. **Integrate with CSS variables** for dynamic theming
4. **Build reusable configuration presets**
5. **Add plugins** for extended functionality

The key is to leverage Tailwind's configuration system rather than writing custom CSS. This maintains consistency, improves performance, and makes your design system more maintainable.

**Remember:** Tailwind is designed to be customized. Don't be afraid to adapt it to your project's needs while maintaining the utility-first philosophy.
